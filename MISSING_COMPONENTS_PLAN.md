**Purpose:**

**Specification:**

- Save as: `brown_bear_matching_level{X}_preview.pdf`
- Include cover + 2 activity pages

### Implementation Plan

**New Script: `generators/marketing/preview_creator.py`**

```python
from PyPDF2 import PdfReader, PdfWriter
from reportlab.pdfgen import canvas
from reportlab.lib.pagesizes import letter
from io import BytesIO

def add_watermark(input_pdf, output_pdf, watermark_text="PREVIEW - © Small Wins Studio"):
    """Add diagonal watermark to PDF pages"""
    reader = PdfReader(input_pdf)
    writer = PdfWriter()
    
    for page_num in range(min(3, len(reader.pages))):  # First 3 pages
        page = reader.pages[page_num]
        
        # Create watermark
        packet = BytesIO()
        can = canvas.Canvas(packet, pagesize=letter)
        can.setFont("Helvetica", 60)
        can.setFillColorRGB(0.5, 0.5, 0.5, alpha=0.3)
        can.translate(4.25*72, 5.5*72)
        can.rotate(45)
        can.drawCentredString(0, 0, watermark_text)
        can.save()
        
        # Merge watermark with page
        packet.seek(0)
        watermark = PdfReader(packet)
        page.merge_page(watermark.pages[0])
        writer.add_page(page)
    
    with open(output_pdf, 'wb') as f:
        writer.write(f)

def create_preview(level, theme="brown_bear", product="matching"):
    """Create preview PDF for TpT"""
    input_pdf = f"outputs/{theme}/{product}/level{level}/product/{theme}_{product}_level{level}_color.pdf"
    output_pdf = f"outputs/{theme}/{product}/level{level}/marketing/{theme}_{product}_level{level}_preview.pdf"
    
    add_watermark(input_pdf, output_pdf)
    print(f"✓ Created preview for Level {level}")
```

```python
from pdf2image import convert_from_path
from PIL import Image

def create_thumbnails(cover_pdf, output_dir):
    """Generate thumbnail images from cover PDF"""
    
    # Convert PDF to image
    images = convert_from_path(cover_pdf, dpi=300)
    cover_image = images[0]
    
    # Generate different sizes
    sizes = {
        'tpt_listing': (280, 280),
        'tpt_detail': (500, 500),
        'social': (1200, 630)
    }
    
    for name, size in sizes.items():
        # Resize maintaining aspect ratio
        thumb = cover_image.copy()
        thumb.thumbnail(size, Image.LANCZOS)
        
        # Save with optimization
        output_file = f"{output_dir}/thumbnail_{name}_{size[0]}x{size[1]}.png"
        thumb.save(output_file, "PNG", optimize=True)
        print(f"✓ Created {name} thumbnail: {size[0]}×{size[1]}")
```
